23 BSpline(size_t knots, T a, T b)
24 : knots(knots),
25 x_min(a),
26 x_max(b),
27 step((b - a) / (knots - 1)),
28 inv_step(T{1} / step) {}
29
30 auto eval(T x) const -> std::vector<T> {
31 std::vector<T> B(knots, T{0});
32
33 if (x <= x_min) {
34 B[0] = T{1};
35 return B;
36 }
37 if (x >= x_max) {
38 B[knots - 1] = T{1};
39 return B;
40 }
41
42 const T pos = (x - x_min) * inv_step;
43 const auto i = static_cast<size_t>(pos);
44 const T t = pos - i;
45
46 B[i] = T{1} - t;
47 B[i + 1] = t;
48 return B;
49 }
